除了一个键，我如何克隆一个 JavaScript 对象？

Arya

2023-02-18 02:06

我有一个平面 JS 对象：

{a: 1, b: 2, c: 3, ..., z:26}

我想克隆除一个元素之外的对象：

{a: 1, c: 3, ..., z:26}

最简单的方法是什么（如果可能，最好使用 es6/7）？



解答

JavaScript 中有一个Destructuring assignment syntax 可以使用

let obj = {a: 1, b: 2, c: 3, z:26};
let {b, ...rest} = obj;

// skips the "Unused variable" warning
let {b: _, ...rest} = obj;

// removes property based on the dynamic key
const dynamicKey = "b";
let {[dynamicKey]: _, ...rest} = obj;

现代浏览器已经开箱即用地支持它。请参阅：JavaScript 运算符：解构赋值：对象中的剩余

对于旧的浏览器版本，可以选择使用Babel来支持解构赋值。它将被转译为：

"use strict";

function _objectWithoutProperties(obj, keys) {
  var target = {};
  for (var i in obj) {
    if (keys.indexOf(i) >= 0) continue;
    if (!Object.prototype.hasOwnProperty.call(obj, i)) continue;
    target[i] = obj[i];
  }
  return target;
}

var x = { a: 1, b: 2, c: 3, z: 26 };
var b = x.b;

var y = _objectWithoutProperties(x, ["b"]);

const obj = { a: 1, b: 2, c: 3, d: 4 }
const clone = (({ b, c, ...o }) => o)(obj) // remove b and c
console.log(clone)
